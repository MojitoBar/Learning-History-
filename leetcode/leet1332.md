class Solution {
    func removePalindromeSub(_ s: String) -> Int {
        return isPalindrome(s) ? 1:2
    }
    
    func isPalindrome(_ s: String) -> Bool {
        let reversedstr = String(s.reversed())
        return s == reversedstr
    }
}
